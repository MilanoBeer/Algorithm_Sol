```python
# BJ_1929_소수 구하기

# isPrime : 2부터 num의 제곱근까지만 검사했을 때 결과로 소수 판별 가능!
def isPrime(num):
    if num == 1:
        return False
    else:
        for i in range(2, int(num ** 0.5) + 1):
            if num % i == 0:
                return False
        return True

M, N = map(int, input().split())
# print(M, N)

# M 이상, N이하의 모든 소수 출력

for i in range(M, N+1):
    if isPrime(i):
        print(i)
```
